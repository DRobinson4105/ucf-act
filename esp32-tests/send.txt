#include <stdio.h>
#include <inttypes.h>
#include "freertos/FreeRTOS.h"
#include "freertos/task.h"
#include "driver/twai.h"
#include "esp_log.h"

static const char *TAG = "CAN_TX";

#define CAN_TX_GPIO 4
#define CAN_RX_GPIO 5

extern "C" void app_main(void)
{
    twai_general_config_t g_config = TWAI_GENERAL_CONFIG_DEFAULT(
        (gpio_num_t)CAN_TX_GPIO,
        (gpio_num_t)CAN_RX_GPIO,
        TWAI_MODE_NORMAL
    );

    twai_timing_config_t t_config = TWAI_TIMING_CONFIG_1MBITS();
    twai_filter_config_t f_config = TWAI_FILTER_CONFIG_ACCEPT_ALL();

    ESP_ERROR_CHECK(twai_driver_install(&g_config, &t_config, &f_config));
    ESP_ERROR_CHECK(twai_start());

    ESP_LOGI(TAG, "TWAI started at 1 Mbit/s in NORMAL mode");

    uint8_t counter = 0;

    while (true) {
        twai_message_t msg = {};
        msg.identifier = 0x123;
        msg.extd = 0;
        msg.rtr = 0;
        msg.data_length_code = 8;

        msg.data[0] = 0x11;
        msg.data[1] = 0x22;
        msg.data[2] = 0x33;
        msg.data[3] = 0x44;
        msg.data[4] = 0x55;
        msg.data[5] = 0x66;
        msg.data[6] = 0x77;
        msg.data[7] = counter++;

        esp_err_t err = twai_transmit(&msg, pdMS_TO_TICKS(1000));
        if (err == ESP_OK) {
            ESP_LOGI(TAG, "Sent 0x%03" PRIX32, msg.identifier);
        } else {
            ESP_LOGE(TAG, "twai_transmit failed: %s", esp_err_to_name(err));
        }

        vTaskDelay(pdMS_TO_TICKS(500));
    }
}
